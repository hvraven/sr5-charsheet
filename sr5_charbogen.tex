\usepackage{calc}
\newlength{\BoxWidth}
\newlength{\BoxSep}
\newlength{\TitleWidth}
\setlength{\BoxWidth}{95mm}
\setlength{\BoxSep}{3mm}
\newcommand{\BoxTitle}[1]{\heading \color{white} \MakeUppercase{#1}}
\newcommand{\BoxRight}[3]{% title, y, height
    \path[fill=border] (\BoxSep,#2) -- node[right] {\BoxTitle{#1}} ++(0mm,-4.5mm)%
        -- ++(\TitleWidth,0mm) -- ++(7.5mm,0mm) -- ++(-7.5mm,4.5mm) -- cycle (\BoxSep,#2);%
    \path[border end] (\BoxSep,#2-#3) -- ++(0mm,12mm);%
    \path[border end] (\BoxSep+\BoxWidth,#2-#3) -- ++(0mm,12mm);%
    \path[border] (\BoxSep,#2) -- ++(0mm,-#3+12mm) ++(\BoxWidth,0mm) --
        ++(0mm,#3-20.5mm) -- ++(-4mm,4mm) -- ++(-\BoxWidth+4mm,0mm);
}
\newcommand{\BoxLeft}[3]{% title, y, height
    \path[fill=border] (-\BoxSep,#2) -- node[left] {\BoxTitle{#1}} ++(0mm,-4.5mm)%
        -- ++(-\TitleWidth,0mm) -- ++(-7.5mm,0mm) -- ++(7.5mm,4.5mm) -- cycle (-\BoxSep,#2);%
    \path[border end] (-\BoxSep,#2-#3) -- ++(0mm,12mm) ++(-\BoxWidth,0mm) -- ++(0mm,-12mm);%
%    \path[border end] (-\BoxSep-\BoxWidth,#2-#3) -- ++(0mm,12mm);%
    \path[border] (-\BoxSep,#2) -- ++(0mm,-#3+12mm) ++(-\BoxWidth,0mm) --
        ++(0mm,#3-20.5mm) -- ++(4mm,4mm) -- ++(\BoxWidth-4mm,0mm);
}
